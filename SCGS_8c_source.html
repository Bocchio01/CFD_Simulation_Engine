163
164 if (k == PP)
165 {

207void CFD_SCGS_System_Solve(SCGS_t *scgs)
208{
209 double DEN;
210 double r1;
211 double r2;
212 double r3;
213 double r4;
214
215 r1 = scgs->vanka->A->data[4][0] / scgs->vanka->A->data[0][0];
216 r2 = scgs->vanka->A->data[4][1] / scgs->vanka->A->data[1][1];
217 r3 = scgs->vanka->A->data[4][2] / scgs->vanka->A->data[2][2];
218 r4 = scgs->vanka->A->data[4][3] / scgs->vanka->A->data[3][3];
219
220 DEN = r1 * scgs->vanka->A->data[0][4] +
221 r2 * scgs->vanka->A->data[1][4] +
222 r3 * scgs->vanka->A->data[2][4] +
223 r4 * scgs->vanka->A->data[3][4];
224
225 scgs->vanka->x->data[4] = (r1 * scgs->vanka->R->data[0] +
226 r2 * scgs->vanka->R->data[1] +
227 r3 * scgs->vanka->R->data[2] +
228 r4 * scgs->vanka->R->data[3] -
229 scgs->vanka->R->data[4]) /
230 DEN;
231
232 scgs->vanka->x->data[0] = (scgs->vanka->R->data[0] - scgs->vanka->A->data[0][4] * scgs->vanka->x->data[4]) / scgs->vanka->A->data[0][0];
233 scgs->vanka->x->data[1] = (scgs->vanka->R->data[1] - scgs->vanka->A->data[1][4] * scgs->vanka->x->data[4]) / scgs->vanka->A->data[1][1];
234 scgs->vanka->x->data[2] = (scgs->vanka->R->data[2] - scgs->vanka->A->data[2][4] * scgs->vanka->x->data[4]) / scgs->vanka->A->data[2][2];
235 scgs->vanka->x->data[3] = (scgs->vanka->R->data[3] - scgs->vanka->A->data[3][4] * scgs->vanka->x->data[4]) / scgs->vanka->A->data[3][3];
236}
