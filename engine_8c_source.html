14void CFD_Setup_Engine(CFD_t *cfd)
15{
16 if (!isnan(cfd->in->fluid->Re))
17 {
18 cfd->in->fluid->nu = fabs(cfd->in->uLid) * cfd->in->geometry->x / cfd->in->fluid->Re;

21 CFD_Setup_Mesh(cfd);

25 log_info("Engine setup complete");
