51static void stdout_callback(log_Event *ev)
52{
53 char buf[16];
54 buf[strftime(buf, sizeof(buf), "%H:%M:%S", ev->time)] = '\0';
55#ifdef LOG_USE_COLOR
56 fprintf(
57 ev->udata, "%s %s%-5s\x1b[0m \x1b[90m%s:%d:\x1b[0m ",
58 buf, level_colors[ev->level], level_strings[ev->level],
59 ev->file, ev->line);
60#else
61 fprintf(
62 ev->udata, "%s %-5s %s:%d: ",
63 buf, level_strings[ev->level], ev->file, ev->line);
64#endif
65 vfprintf(ev->udata, ev->fmt, ev->ap);
66 fprintf(ev->udata, "\n");
67 fflush(ev->udata);
68}

119int log_add_callback(log_LogFn fn, void *udata, int level)
120{
121 for (int i = 0; i < MAX_CALLBACKS; i++)
122 {
123 if (!L.callbacks[i].fn)
124 {
125 L.callbacks[i] = (Callback){fn, udata, level};
126 return 0;
127 }
128 }
129 return -1;
130}

147void log_log(int level, const char *file, int line, const char *fmt, ...)
148{
149 log_Event ev = {
150 .fmt = fmt,
151 .file = file,
152 .line = line,
153 .level = level,
154 };
155
156 lock();
157
158 if (!L.quiet && level >= L.level)
159 {
160 init_event(&ev, stderr);
161 va_start(ev.ap, fmt);
162 stdout_callback(&ev);
163 va_end(ev.ap);
164 }
165
166 for (int i = 0; i < MAX_CALLBACKS && L.callbacks[i].fn; i++)
167 {
168 Callback *cb = &L.callbacks[i];
169 if (level >= cb->level)
170 {
171 init_event(&ev, cb->udata);
172 va_start(ev.ap, fmt);
173 cb->fn(&ev);
174 va_end(ev.ap);
175 }
176 }
177
178 unlock();
179}
